/************************************************************* * 용접 안전 점검 시스템 - 구글 시트 연동 백엔드 (무료) * --------------------------------------------------------- * [사용법] * 1) 구글 드라이브에서 새 [구글 스프레드시트] 1개를 만든다. * 2) 상단 메뉴 [확장 프로그램] > [Apps Script] 클릭. * 3) 기본 코드를 전부 지우고 이 파일 내용을 그대로 붙여넣는다. * 4) 상단 [배포] > [새 배포] > 유형 '웹 앱' 선택 * - 실행 계정: 나 * - 액세스 권한: '모든 사용자'(익명 포함) * [배포] 클릭 후 나오는 '웹 앱 URL'(.../exec)을 복사한다. * 5) 용접안전점검시스템 HTML 파일 안의 * CONFIG = { GAS_URL: 'https://script.google.com/macros/s/AKfycbxFsjmey7KbZBtCtcgEi4Zg-264L0TA_l9xAJRI0ZH-2kc4qkOH7J4Em-HPWoRblIXvow/exec' } * 따옴표 안에 복사한 URL을 붙여넣는다. * 끝! 이제 학생이 제출할 때마다 날짜별 시트에 자동 기록되고, * 사진/서명은 드라이브에 저장된 뒤 링크가 시트에 들어갑니다. *************************************************************/ // 사진·서명 이미지를 저장할 드라이브 폴더 이름 var DRIVE_FOLDER = '용접안전점검_증빙자료'; function doPost(e) { var lock = LockService.getScriptLock(); lock.waitLock(30000); try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); // 날짜(YYYY-MM-DD)별 탭(시트)을 찾거나 새로 만든다 — '일별' 정리 var tabName = data.date || todayStr_(); var sheet = ss.getSheetByName(tabName); if (!sheet) { sheet = ss.insertSheet(tabName); buildHeader_(sheet, tabName); } // 사진/서명을 드라이브에 저장하고 링크 생성 var folder = getDateFolder_(tabName); var prefix = data.cls + '_' + data.name; var photo1 = saveImage_(folder, data.photo1, prefix + '_사진1'); var photo2 = saveImage_(folder, data.photo2, prefix + '_사진2'); var sigUrl = saveImage_(folder, data.sig, prefix + '_서명'); sheet.appendRow([ data.time || '', data.cls || '', data.name || '', data.bangyeom || '', data.safetyShoes || '', data.gears || '', data.result || '', photo1, photo2, sigUrl ]); // 결과(이상없음/미비있음) 셀 색상 강조 var lastRow = sheet.getLastRow(); var resCell = sheet.getRange(lastRow, 7); resCell.setBackground(data.result === '이상없음' ? '#D8EFD3' : '#FBE0DF') .setFontColor(data.result === '이상없음' ? '#1E6B3A' : '#A32D2D') .setFontWeight('bold'); return json_({ ok: true, sheet: tabName }); } catch (err) { return json_({ ok: false, error: String(err) }); } finally { lock.releaseLock(); } } // 브라우저로 URL을 직접 열었을 때 동작 확인용 function doGet() { return json_({ ok: true, message: '용접 안전 점검 백엔드 정상 작동 중' }); } /* ---------- 보조 함수 ---------- */ // 공문서 형식의 머리글(제목 + 열 이름) function buildHeader_(sheet, tabName) { sheet.getRange('A1:J1').merge() .setValue('용접 실습 안전 점검표 · ' + tabName + ' · 스마트용접과') .setFontSize(14).setFontWeight('bold') .setHorizontalAlignment('center').setVerticalAlignment('middle') .setBackground('#0F6E56').setFontColor('#FFFFFF'); sheet.setRowHeight(1, 38); var headers = ['제출시각', '학년·반', '이름', '방염복', '안전화', '보호장구(헬멧/두건/각반/토시/앞치마)', '판정', '착용사진1', '착용사진2', '자필서명']; var hr = sheet.getRange(2, 1, 1, headers.length); hr.setValues([headers]) .setFontWeight('bold').setBackground('#EAF3DE').setFontColor('#1A1D23') .setHorizontalAlignment('center').setBorder(true, true, true, true, true, true); sheet.setColumnWidth(1, 150); sheet.setColumnWidth(2, 130); sheet.setColumnWidth(3, 80); sheet.setColumnWidth(6, 300); sheet.setColumnWidths(8, 3, 110); sheet.setFrozenRows(2); } // 날짜별 드라이브 하위 폴더 가져오기/생성 function getDateFolder_(tabName) { var root = getOrCreateFolder_(DriveApp.getRootFolder(), DRIVE_FOLDER); return getOrCreateFolder_(root, tabName); } function getOrCreateFolder_(parent, name) { var it = parent.getFoldersByName(name); return it.hasNext() ? it.next() : parent.createFolder(name); } // base64 dataURL 이미지를 드라이브에 저장하고 공유 링크 반환 function saveImage_(folder, dataUrl, fileName) { if (!dataUrl || dataUrl.indexOf('base64,') < 0) return ''; try { var parts = dataUrl.split(','); var meta = parts[0]; // data:image/png;base64 var contentType = meta.substring(meta.indexOf(':') + 1, meta.indexOf(';')); var ext = contentType.indexOf('png') >= 0 ? '.png' : '.jpg'; var bytes = Utilities.base64Decode(parts[1]); var blob = Utilities.newBlob(bytes, contentType, fileName + ext); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return file.getUrl(); } catch (err) { return '저장실패'; } } function todayStr_() { var d = new Date(); return Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } function json_(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
